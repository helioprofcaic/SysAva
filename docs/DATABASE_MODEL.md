# 🗄️ Modelo do Banco de Dados (SQL)

Este arquivo contém todos os comandos SQL necessários para criar a estrutura completa do banco de dados do SysAva no Supabase.

Copie e cole todo o bloco de código abaixo no **SQL Editor** do seu projeto Supabase e execute-o uma única vez.

```sql
-- Tabela para armazenar usuários e senhas
create table app_users (
  username text primary key,
  password text not null, -- Armazenará a senha criptografada (hash)
  ra text unique, -- Registro do Aluno
  name text,
  role text default 'student' -- Papel do usuário: 'student' ou 'admin'
);

-- Tabela para as escolas
create table schools (
    id bigint generated by default as identity primary key,
    name text not null,
    gre text
);

-- Tabela para as turmas
create table classes (
    id bigint generated by default as identity primary key,
    code text unique not null, -- Código da Turma
    name text not null,
    school_id bigint references schools(id)
);

-- Tabela para as disciplinas
create table subjects (
    id bigint generated by default as identity primary key,
    name text unique not null
);

-- Tabela de associação para vincular disciplinas a turmas (Muitos-para-Muitos)
create table class_subjects (
    class_id bigint references classes(id) on delete cascade,
    subject_id bigint references subjects(id) on delete cascade,
    primary key (class_id, subject_id)
);

-- Tabela para matricular alunos em turmas
create table student_enrollments (
    user_username text references app_users(username) on delete cascade,
    class_id bigint references classes(id) on delete cascade,
    primary key (user_username, class_id)
);

-- Tabela para as aulas
create table lessons (
    id bigint generated by default as identity primary key,
    title text not null,
    description text,
    video_url text
);

-- Tabela para os quizzes
create table quizzes (
    id bigint generated by default as identity primary key,
    lesson_id bigint references lessons(id) on delete cascade,
    title text not null
);

-- Tabela para as questões dos quizzes
create table quiz_questions (
    id bigint generated by default as identity primary key,
    quiz_id bigint references quizzes(id) on delete cascade,
    question_text text not null,
    options jsonb not null, -- Ex: '["Opção A", "Opção B"]'
    correct_option_index int not null -- Ex: 0 para a primeira opção
);

-- Tabela para armazenar mensagens do fórum
create table forum_posts (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  user_name text,
  message text,
  lesson_id bigint references lessons(id)
);
```
