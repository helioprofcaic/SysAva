# 🗄️ Modelo do Banco de Dados (SQL)

Este arquivo contém todos os comandos SQL necessários para criar a estrutura completa do banco de dados do SysAva no Supabase.

Copie e cole todo o bloco de código abaixo no **SQL Editor** do seu projeto Supabase e execute-o uma única vez.

```sql
-- =============================================================================
-- 🛠️ SETUP DO BANCO DE DADOS (SUPABASE / POSTGRESQL)
-- =============================================================================
-- Este script cria toda a estrutura de tabelas para o SysAva.
-- É seguro executá-lo múltiplas vezes, pois ele usa "CREATE TABLE IF NOT EXISTS".

-- 1. Tabela de Usuários
CREATE TABLE IF NOT EXISTS public.app_users (
    username TEXT PRIMARY KEY,
    password TEXT NOT NULL,
    name TEXT NOT NULL,
    ra TEXT UNIQUE,
    role TEXT DEFAULT 'student'
);

-- 2. Estrutura Acadêmica (Escolas, Turmas, Disciplinas)
CREATE TABLE IF NOT EXISTS public.schools (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    gre TEXT
);

CREATE TABLE IF NOT EXISTS public.classes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    code TEXT UNIQUE NOT NULL,
    school_id BIGINT REFERENCES public.schools(id)
);

CREATE TABLE IF NOT EXISTS public.subjects (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT UNIQUE NOT NULL
);

-- Tabela de junção: Turmas <-> Disciplinas
CREATE TABLE IF NOT EXISTS public.class_subjects (
    class_id BIGINT REFERENCES public.classes(id) ON DELETE CASCADE,
    subject_id BIGINT REFERENCES public.subjects(id) ON DELETE CASCADE,
    PRIMARY KEY (class_id, subject_id)
);

-- Matrículas dos alunos
CREATE TABLE IF NOT EXISTS public.student_enrollments (
    user_username TEXT REFERENCES public.app_users(username) ON DELETE CASCADE,
    class_id BIGINT REFERENCES public.classes(id) ON DELETE CASCADE,
    PRIMARY KEY (user_username, class_id)
);

-- 3. Conteúdo de Ensino (Aulas, Fórum, Quiz)
CREATE TABLE IF NOT EXISTS public.lessons (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    video_url TEXT,
    subject_id BIGINT REFERENCES public.subjects(id),
    -- Garante que não existam duas aulas com mesmo título na mesma disciplina
    CONSTRAINT lessons_subject_id_title_key UNIQUE (subject_id, title)
);

CREATE TABLE IF NOT EXISTS public.forum_posts (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_name TEXT NOT NULL,
    message TEXT NOT NULL,
    lesson_id BIGINT REFERENCES public.lessons(id) ON DELETE CASCADE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.quizzes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    lesson_id BIGINT REFERENCES public.lessons(id) ON DELETE CASCADE,
    title TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS public.quiz_questions (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    quiz_id BIGINT REFERENCES public.quizzes(id) ON DELETE CASCADE,
    question_text TEXT NOT NULL,
    options JSONB NOT NULL, -- Ex: ["Opção A", "Opção B", "Opção C"]
    correct_option_index INTEGER NOT NULL
);

-- =============================================================================
-- 🧹 COMANDOS ÚTEIS (MANUTENÇÃO)
-- =============================================================================

```
